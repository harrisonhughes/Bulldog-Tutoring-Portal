#Database Creation
CREATE DATABASE tutoring_portal
CHARACTER SET utf8;

USE tutoring_portal;

#Table of User Accounts
CREATE TABLE accounts(
    email VARCHAR(50) PRIMARY KEY,
    hashed_password CHAR(64) NOT NULL,
    firstname VARCHAR(50) NOT NULL,
    lastname VARCHAR(50) NOT NULL,
   	account_type TINYINT(1) DEFAULT 0,
    last_activity TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

#Table of Classes Offered
CREATE TABLE courses(
    id INT(3) AUTO_INCREMENT PRIMARY KEY,
    subject VARCHAR(5) NOT NULL,
    course_code VARCHAR(5) NOT NULL
);

#Table of Active Tutors
CREATE TABLE active_tutors(
    email VARCHAR(50) NOT NULL,
    course_id INT(3) NOT NULL,
    PRIMARY KEY (email, course_id),
    FOREIGN KEY (email) REFERENCES accounts(email) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE
);

#Table of Referred Tutors
CREATE TABLE referred_tutors(
    email VARCHAR(50) NOT NULL,
    course_id INT(3) NOT NULL,
    PRIMARY KEY (email, course_id),
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE
);

#Table of Professor/Course pairs for a semester
CREATE TABLE course_professors(
    email VARCHAR(50) NOT NULL,
    course_id INT(3) NOT NULL,
    PRIMARY KEY (email, course_id),
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE
);

#List of email accounts that have opted out of email receiving 
CREATE TABLE opt_outs(
    email VARCHAR(50) PRIMARY KEY,
    date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

#Table to keep track of admin transition period information
CREATE TABLE semesters(
    term_code INT(6) PRIMARY KEY,
    file_uploaded TIMESTAMP DEFAULT "2000-01-01 00:00:00",
    professor_emails TIMESTAMP DEFAULT "2000-01-01 00:00:00",
    student_emails TIMESTAMP DEFAULT "2000-01-01 00:00:00",
    professor_reminders INT(1) DEFAULT 0,
    student_reminders INT(1) DEFAULT 0
);
